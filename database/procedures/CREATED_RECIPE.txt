CREATE OR REPLACE PROCEDURE "CREATED_RECIPE"(ID IN NUMBER, STATUS OUT VARCHAR2, INFO_RC OUT SYS_REFCURSOR) AS
BEGIN
	-- routine body goes here, e.g.
	-- DBMS_OUTPUT.PUT_LINE('Navicat for Oracle');
	OPEN INFO_RC FOR 
	SELECT R.RECIPE_ID, R.TITLE
FROM USERS U LEFT JOIN RECIPES R ON R.USER_ID = U.USER_ID
WHERE U.USER_ID = ID;
STATUS := 'SUCCESSFUL';
EXCEPTION
	WHEN NO_DATA_FOUND THEN
	STATUS := 'NO DATA FOUND';
	WHEN OTHERS THEN
	STATUS := 'UNKNOWN ERROR';
END;
